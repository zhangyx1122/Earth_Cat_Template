    int x, y, z;
};

    scanf("%d", &n);
    for (i = 0; i < n; ++i)
        for (j = i + 1; j < n; ++j) {
            int k = (2 * n - i - j) % n;
            if (k > j)

}

n = int(input())


def gen_table(m):
    table = [[(i + j) % m for j in range(m)] for i in range(m)]

    d = dict()
    for i in range(m):
        d[table[i][i]] = i
    for i in range(m):
        for j in range(m):
            table[i][j] = d[table[i][j]]

    return table


def gen_table2(m):
    assert (m % 2 == 0)
    table = [[(i + j) % m for j in range(m)] for i in range(m)]

    d = {i: i for i in range(m)}
    for i in range(m // 2):
        d[i], d[table[i][i]] = d[table[i][i]], d[i]

    d = {v: u for u, v in d.items()}

    for i in range(m):
        for j in range(m):
            table[i][j] = d[table[i][j]]

    return table


def skolem(n):
    assert (n % 6 == 1)
    ans = []
    m = n // 3
    table = gen_table2(m)

    inf = n
    mm = m // 2

    for i in range(1, mm + 1):
        u = i
        v = i + m
        w = i + m * 2
        ans.append((u, v, w))

    for k in range(3):
        for i in range(1, mm + 1):
            u = mm + i + k * m
            v = i + (k + 1) % 3 * m
            w = inf
            u, v, w = sorted([u, v, w])
            ans.append((u, v, w))

    for k in range(3):
        for i in range(1, m + 1):
            for j in range(i + 1, m + 1):
                u = i + k * m
                v = j + k * m
                w = table[i - 1][j - 1] + 1 + (k + 1) % 3 * m
                u, v, w = sorted([u, v, w])
                ans.append((u, v, w))

    return ans


def bose(n):
    assert (n % 6 == 3)
    ans = []
    m = n // 3
    table = gen_table(m)

    for i in range(1, m + 1):
        u = i
        v = i + m
        w = i + m * 2
        ans.append((u, v, w))

    for k in range(3):
        for i in range(1, m + 1):
            for j in range(i + 1, m + 1):
                u = i + k * m
                v = j + k * m
                w = table[i - 1][j - 1] + 1 + (k + 1) % 3 * m
                u, v, w = sorted([u, v, w])
                ans.append((u, v, w))

    return ans


def five(n):
    assert (n % 6 == 5)
    ans = []
    m = n // 3
    table = gen_table(m)

    inf1 = 3 * m + 1
    inf2 = 3 * m + 2

    def alpha(num):
        if num == 1: return 1
        num += 1
        if num > m:
            num = 2
        return num

    def beta(num):
        if num == 1: return 1
        num -= 1
        if num == 1:
            num = m
        return num

    mm = m // 2
    for i in range(1, mm + 1):
        u = 2 * i
        v = 2 * i + m
        w = inf1
        u, v, w = sorted([u, v, w])
        ans.append((u, v, w))

        u = 2 * i + 2 * m
        v = alpha(2 * i)
        w = inf1
        u, v, w = sorted([u, v, w])
        ans.append((u, v, w))

        u = alpha(2 * i) + m
        v = alpha(2 * i) + 2 * m
        w = inf1
        u, v, w = sorted([u, v, w])
        ans.append((u, v, w))

        u = 2 * i + m
        v = 2 * i + 2 * m
        w = inf2
        u, v, w = sorted([u, v, w])
        ans.append((u, v, w))

        u = alpha(2 * i)
        v = alpha(2 * i) + m
        w = inf2
        u, v, w = sorted([u, v, w])
        ans.append((u, v, w))

        u = 2 * i
        v = beta(2 * i) + 2 * m
        w = inf2
        u, v, w = sorted([u, v, w])
        ans.append((u, v, w))

    for i in range(1, m + 1):
        for j in range(i + 1, m + 1):
            u = i
            v = j
            w = table[i - 1][j - 1] + 1 + m
            u, v, w = sorted([u, v, w])
            ans.append((u, v, w))

            u = i + m
            v = j + m
            w = table[i - 1][j - 1] + 1 + 2 * m
            u, v, w = sorted([u, v, w])
            ans.append((u, v, w))

            u = i + 2 * m
            v = j + 2 * m
            w = alpha(table[i - 1][j - 1] + 1)
            u, v, w = sorted([u, v, w])
            ans.append((u, v, w))

    ans.append((1, 1 + m, 1 + 2 * m))
    ans.append((1, inf1, inf2))

    return ans


if n % 6 == 1:  # Skolem construction
    ans = skolem(n)
elif n % 6 == 2:  # Skolem construction of n - 1
    ans = skolem(n - 1)
elif n % 6 == 3:  # Bose construction
    ans = bose(n)
elif n % 6 == 4:  # Bose construction of n - 1
    ans = bose(n - 1)
elif n % 6 == 5:
    ans = five(n)
else:
    raw_ans = skolem(n + 1)
    ans = []
    for u, v, w in raw_ans:
        if w <= n:
            ans.append((u, v, w))

print(len(ans))
for u, v, w in ans:
    print(u, v, w)
