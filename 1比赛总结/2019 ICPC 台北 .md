## J - Automatic Control Machine 

**题意** ： 有n个任务，m种操作，问最少的能把所有任务完成的操作数

$T \le 10, n \le 500, m \le 15$

"这不舞蹈链吗"

"m挺小的"

纯暴力代码

```cpp
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
const ll N = 1 << 20;
const ll mod = 1e9+7;
const ll inf = 0x3f3f3f3f3f3f3f3f;

int shu(int x){
    int cnt=0;
    while(x){
        if(x&1) cnt++;
        x>>=1;
    }
    return cnt;
}

char a[20][505];
int main(){
    ios::sync_with_stdio(0);
    int t;
    cin>>t;
    while(t--){
        int n,m;
        cin>>m>>n;
        for(int i=0;i<n;i++) cin>>a[i];
        int mmin=0x3f3f3f3f;
        for(int i=1;i<(1<<n);i++){
            bool ans[505];
            for(int k=0;k<=m;k++) ans[k]=0;
            for(int j=0;j<n;j++){
                if(i&(1<<j)) {
                    for(int k=0;k<m;k++){
                        if(a[j][k]=='1') ans[k]=1;
                    }
                }
            }
            bool ok=1;
            for(int j=0;j<m;j++) if(ans[j]==0) ok=0;
            if(ok) mmin=min(mmin,shu(i));
        }
        if(mmin==0x3f3f3f3f){
            cout<<-1<<endl;
            continue;
        }
        cout<<mmin<<endl;
    }
}
```

		{
			scanf("%s",str);
			number[i]=bitset<500>(str);
		}
		int len=1<<m,ans=m+1;
		for(int i=1;i<len;i++)
		{
			int t=i,s=0;
			bitset<500> num(0);
			for(int j=0;j<m&&t>0;j++)
			{
				if(t&1) 
				{
					num=num|number[j];
					s++;
				}
				t>>=1;
			}
			if(num.count()==n) ans=min(ans,s);
		}
		if(ans==m+1) printf("-1\n");
		else printf("%d\n",ans);
	}
	return 0;
}

## A - Rush Hour Puzzle

**题意** : $6 \times 6$ 的格子上有很多车（最多100），每辆车长度为2或3，问要把目标车辆移出来的最小步数，

因为超过10就当做不能移出，暴力

10层讨论发现每辆车只会往一个方向开

刚开始题目没有交接清楚，以为可以向四个方向移动

```cpp
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
const ll N = 1 << 20;
const ll mod = 1e9+7;
const ll inf = 0x3f3f3f3f3f3f3f3f;

int a[10][10];
int n=0;
vector<int> g[12];

struct car{
    pair<int,int> pos1,pos2;
    bool dir;
}c[12];


void dfs(int u,bool *vis){
    vis[u]=1;
    if(c[u].dir==1){
        int x=c[u].pos1.first+1;
        int y=c[u].pos1.second;
        if(x<=6&&a[x][y]&&!vis[a[x][y]]){
            dfs(a[x][y],vis);
        }
        x=c[u].pos2.first-1;
        y=c[u].pos2.second;
        if(x>0&&a[x][y]&&!vis[a[x][y]]){
            dfs(a[x][y],vis);
        }
    }else {
        int x=c[u].pos1.first;
        int y=c[u].pos1.second+1;
        if(y<=6&&a[x][y]&&!vis[a[x][y]]){
            dfs(a[x][y],vis);
        }
        x=c[u].pos2.first;
        y=c[u].pos2.second-1;
        if(y>0&&a[x][y]&&!vis[a[x][y]]){
            dfs(a[x][y],vis);
        }
    }
}

bool checkp(int u){
    if(c[u].dir==1){
        int x=c[u].pos1.first+1;
        int y=c[u].pos1.second;
        if(x<=6&&a[x][y]==0) return 1;
    }else {
        int x=c[u].pos1.first;
        int y=c[u].pos1.second+1;
        if(y<=6&&a[x][y]==0) return 1;
    }
    return 0;
}

bool checks(int u){
    if(c[u].dir==1){
        int x=c[u].pos2.first-1;
        int y=c[u].pos2.second;
        if(x>0&&a[x][y]==0) return 1;
    }else {
        int x=c[u].pos2.first;
        int y=c[u].pos2.second-1;
        if(y>0&&a[x][y]==0) return 1;
    }
    return 0;
}

bool succeed(){
    if(c[1].dir==1){
        if(c[1].pos1.first==6) return 1;
    }else {
        if(c[1].pos1.second==6) return 1;
    }
    return 0;
}

void movep(int u){
    if(c[u].dir==1){
        int x=c[u].pos1.first+1;
        int y=c[u].pos1.second;
        a[x][y]=u;
        int rx=c[u].pos2.first;
        int ry=c[u].pos2.second;
        a[rx][ry]=0;
        c[u].pos1.first++;
        c[u].pos2.first++;
    }else {
        int x=c[u].pos1.first;
        int y=c[u].pos1.second+1;
        a[x][y]=u;
        int rx=c[u].pos2.first;
        int ry=c[u].pos2.second;
        a[rx][ry]=0;
        c[u].pos1.second++;
        c[u].pos2.second++;
    }
}

void moves(int u){
    if(c[u].dir==1){
        int x=c[u].pos2.first-1;
        int y=c[u].pos2.second;
        a[x][y]=u;
        int rx=c[u].pos1.first;
        int ry=c[u].pos1.second;
        a[rx][ry]=0;
        c[u].pos1.first--;
        c[u].pos2.first--;
    }else {
        int x=c[u].pos2.first;
        int y=c[u].pos2.second-1;
        a[x][y]=u;
        int rx=c[u].pos1.first;
        int ry=c[u].pos1.second;
        a[rx][ry]=0;
        c[u].pos1.second--;
        c[u].pos2.second--;
    }
}

void getConflict(bool *vis){
    for(int i=0;i<12;i++) vis[i]=0;
    dfs(1,vis);
}
int ans=0x3f3f3f3f;
void solve(int d){
    if(succeed()){
        ans=min(ans,d+2);
        return;
    }
    if(d>=8) return;
    bool vis[12];
    getConflict(vis);
    for(int i=1;i<=n;i++){
        if(!vis[i]) continue;
        if(checkp(i)){
            movep(i);
            solve(d+1);
            moves(i);
        }
        if(checks(i)){
            moves(i);
            solve(d+1);
            movep(i);
        }
    }
}

int main(){
    ios::sync_with_stdio(0);
    for(int i=1;i<12;i++) c[i].pos2=pair<int,int>(0x3f3f3f3f,0x3f3f3f3f);
    for(int i=1;i<=6;i++){
        for(int j=1;j<=6;j++){
            cin>>a[i][j];
            n=max(a[i][j],n);
            if(a[i][j]){
                c[a[i][j]].pos1=max(c[a[i][j]].pos1,pair<int,int>(i,j));
                c[a[i][j]].pos2=min(c[a[i][j]].pos2,pair<int,int>(i,j));
            }
        }
    }
    for(int i=1;i<=n;i++){
        if(c[i].pos1.first==c[i].pos2.first) c[i].dir=0;
        else c[i].dir=1;
    }
    solve(0);
    if(ans==0x3f3f3f3f) cout<<-1<<endl;
    else cout<<ans<<endl;
}

```
